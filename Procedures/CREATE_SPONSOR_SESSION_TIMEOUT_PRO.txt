DROP PROCEDURE IF EXISTS abi_newadmin.CREATE_SPONSOR_SESSION_TIMEOUT_PRO;
CREATE PROCEDURE abi_newadmin.`CREATE_SPONSOR_SESSION_TIMEOUT_PRO`(
 IN P_SPONSOR_ID INTEGER,
 IN P_TIME_IN DATETIME,
 IN P_TIME_OUT DATETIME
 )
BEGIN
 
   /* Exception handling to roll back */
DECLARE EXIT HANDLER FOR SQLEXCEPTION, SQLWARNING
    BEGIN        
        ROLLBACK;         
    END;
    
 INSERT INTO SPONSOR_SESSION_TIMEOUT
 ( SPONSOR_ID,TIME_IN,TIME_OUT)
 VALUES
 (P_SPONSOR_ID,P_TIME_IN,P_TIME_OUT);
 
 COMMIT;
 
 
 
 END;
