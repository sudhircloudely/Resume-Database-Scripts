DROP PROCEDURE IF EXISTS abi_newadmin.CREATE_SPONSOR_RESUME_VIEW_HISTORY_PRO;
CREATE PROCEDURE abi_newadmin.`CREATE_SPONSOR_RESUME_VIEW_HISTORY_PRO`(
 IN P_SPONSOR_ID INTEGER,
 IN P_RESUME_ID INTEGER,    
 IN P_CREATED_BY varchar(255) )
BEGIN 
 
  /* Exception handling to roll back */
DECLARE EXIT HANDLER FOR SQLEXCEPTION, SQLWARNING
    BEGIN        
        ROLLBACK;         
    END;
 
   INSERT INTO SPONSOR_RESUME_VIEW_HISTORY
   (SPONSOR_ID,RESUME_ID,CREATED_BY,CREATED_DATE)
   VALUES
   (P_SPONSOR_ID,P_RESUME_ID,P_CREATED_BY,NOW() );
        
    
 COMMIT;
 
 END;
